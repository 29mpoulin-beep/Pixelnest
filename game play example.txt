    function playGame() {
      const code = document.getElementById("gameCode").value;
      if (!code.trim()) {
        alert("Paste your game HTML first!");
        return;
      }

      const gameWindow = window.open("", "_blank");
      gameWindow.document.open();
      gameWindow.document.write(code);
      gameWindow.document.close();
    }
